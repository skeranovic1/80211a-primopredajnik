24def oznaci_okvir(ax, up_factor, num_ofdm_symbols, fs):
25 STS_len = 16 * up_factor * 10
26 LTS_len = 160 * up_factor
27 OFDM_symbol_len = 64 * up_factor
28 GI_len = 16 * up_factor
29
30 sts_end = STS_len / fs
31 lts_end = (STS_len + LTS_len) / fs
32 payload_end = (STS_len + LTS_len +
33 num_ofdm_symbols * (OFDM_symbol_len + GI_len)) / fs
34
35 ax.axvline(sts_end * 1e6, color='green', linestyle='--', label='Kraj STS')
36 ax.axvline(lts_end * 1e6, color='orange', linestyle='--', label='Kraj LTS')
37 ax.axvline(payload_end * 1e6, color='purple', linestyle='--', label='Kraj payload')
38
39 for i in range(num_ofdm_symbols):
40 gi_start = (STS_len + LTS_len +
41 i * (GI_len + OFDM_symbol_len)) / fs
42 gi_end = gi_start + GI_len / fs
43 ax.axvspan(gi_start * 1e6, gi_end * 1e6,
44 color='green', alpha=0.25)
